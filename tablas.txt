CREATE TABLE Anunciante(ID_anunciante char(9), nombre varchar(30), sitio_web varchar(30), datos_fiscales varchar(30), PRIMARY KEY (ID_anunciante));
CREATE TABLE Medio(ID_medio char(9), nombre varchar(30), sitio_web varchar(30), datos_fiscales varchar(30), PRIMARY KEY (ID_medio));
CREATE TABLE Anuncio(ID_anuncio char(9),ida REFERENCES Anunciante(ID_anunciante),contenido varchar(1000), url varchar(1000), PRIMARY KEY (ID_anuncio,ida));
CREATE TABLE Espacio(ID_espacio char(9),idm REFERENCES Medio(ID_medio),multimedia varchar(1000), tamano int,enlace varchar(1000), PRIMARY KEY (ID_espacio,idm));
CREATE TABLE Audiencia(ID_audiencia char(9), nombre varchar(30), descripcion varchar(30), PRIMARY KEY (ID_audiencia));
CREATE TABLE Contrato_anunciante(ID_contrato_anunciante char(9), ida REFERENCES Anunciante(ID_anunciante), fecha date, descripcion varchar(30), PRIMARY KEY (ID_contrato_anunciante,ida));
CREATE TABLE Cobro(ID_cobro char(9),idca REFERENCES Contrato_anunciante(ID_contrato_anunciante),ida REFERENCES Anunciante(ID_anunciante),importe real, fecha date,PRIMARY KEY(ID_cobro));
CREATE TABLE Contrato_medio(ID_contrato_medio char(9), idm REFERENCES Medio(ID_medio), fecha date, duracion real ,importe real ,descripcion varchar(30), PRIMARY KEY (ID_contrato_medio,idm));
CREATE TABLE Pago(ID_pago char(9),idcm REFERENCES Contrato_medio(ID_contrato_medio), idm REFERENCES Medio(ID_medio),importe real, fecha date, PRIMARY KEY (ID_pago ));
CREATE TABLE Se_asigna(ide REFERENCES Espacio(ID_espacio), idm REFERENCES Medio(ID_medio),idanu REFERENCES Anuncio(ID_anuncio) ,ida REFERENCES Anunciante(ID_anunciante),PRIMARY KEY (ide,idm));
CREATE TABLE Interesa(ida REFERENCES Anunciante(ID_anunciante),idau REFERENCES Audiencia(ID_audencia),PRIMARY KEY (ida,idau));
CREATE TABLE Visita(idm REFERENCES Medio(ID_medio),idau REFERENCES Audiencia(ID_audencia),PRIMARY KEY (idau,idm));
